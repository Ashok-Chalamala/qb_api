import os
import webbrowser
from google_auth_oauthlib.flow import InstalledAppFlow
import requests

# 1. SETUP CONFIGURATION
# Download your client secret JSON from Google Cloud Console and point to it here
CLIENT_SECRETS_FILE = "client_secret.json"

# Define the Google Health scopes you need
SCOPES = [
    'https://googleapis.com',
    'https://googleapis.com'
]

def get_authenticated_session():
    """Handles OAuth2 web flow using Google's official libraries."""
    flow = InstalledAppFlow.from_client_secrets_file(
        CLIENT_SECRETS_FILE, 
        scopes=SCOPES,
        redirect_uri="http://127.0.0"
    )
    
    # This automatically spins up a local server on port 8080, opens the browser,
    # and intercepts the callback token from Google.
    credentials = flow.run_local_server(port=8080, prompt='consent')
    
    print("Successfully authenticated with Google Health API!")
    return credentials.token

def fetch_health_data(access_token):
    """Queries the new Google Health API unified endpoints."""
    # Google Health consolidated old endpoints into uniform resource shapes
    url = "https://googleapis.com"
    
    headers = {
        "Authorization": f"Bearer {access_token}",
        "Accept": "application/json"
    }
    
    # Request parameters for the new filter syntax
    params = {
        "startTime": "2026-07-01T00:00:00Z",
        "endTime": "2026-07-07T23:59:59Z"
    }
    
    response = requests.get(url, headers=headers, params=params)
    
    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error {response.status_code}: {response.text}")
        return None

def main():
    if not os.path.exists(CLIENT_SECRETS_FILE):
        print(f"Please place your '{CLIENT_SECRETS_FILE}' file from Google Cloud Console in this directory.")
        return

    # Authenticate and grab the token
    access_token = get_authenticated_session()
    
    # Fetch step details
    print("\nFetching step aggregates from Google Health...")
    data = fetch_health_data(access_token)
    if data:
        print(data)

if __name__ == "__main__":
    main()

import requests

def fetch_sleep_data(access_token, start_time, end_time):
    """Fetches sleep sessions, stages (Deep, Light, REM), and durations."""
    url = "https://googleapis.com"
    
    headers = {
        "Authorization": f"Bearer {access_token}",
        "Accept": "application/json"
    }
    
    params = {
        "startTime": start_time,  # Format: "2026-07-01T00:00:00Z"
        "endTime": end_time
    }
    
    response = requests.get(url, headers=headers, params=params)
    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error fetching sleep: {response.status_code} - {response.text}")
        return None


def fetch_heart_rate_data(access_token, start_time, end_time):
    """Fetches heart rate samples (BPM) over a specified time window."""
    # Using 'series' gives you granular timeline beats, 'rollUp' gives averages
    url = "https://googleapis.com"
    
    headers = {
        "Authorization": f"Bearer {access_token}",
        "Accept": "application/json"
    }
    
    params = {
        "startTime": start_time,
        "endTime": end_time
    }
    
    response = requests.get(url, headers=headers, params=params)
    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error fetching heart rate: {response.status_code} - {response.text}")
        return None


Reading the JSON OutputBecause the API is unified, the structure of the data you get back is highly organized:Sleep Output: Returns an array of sleep entries. Each entry contains a startTime and endTime, alongside an internal stage array marking exactly when the user was in REM, Light, Deep, or Awake states.Heart Rate Output: Returns an array of timestamps coupled with a value field containing the BPM (Beats Per Minute) registered at that specific millisecond.
